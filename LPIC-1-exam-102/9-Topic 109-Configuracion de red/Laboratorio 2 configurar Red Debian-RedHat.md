##En redhat:
ifconfig -a
ip a show 
ip a show ens37

nmcli connection add con-name ens37 type ethernet ifname ens37 ipv4.addresses 192.168.0.50/24 ipv4.gateway 192.168.0.1 ipv4.method manual


vi /etc/sysconfig/network-scripts/ifcfg-ens37
TYPE=Ethernet
BOOTPROTO=none
DEFROUTE=yes
NAME=ens37
ONBOOT=yes
IPADDR=192.168.0.50
NETMASK=255.255.255.0
GATEWAY=192.168.0.1

systemctl restart NetworkManager
----------------------------------------------------------
hostnamectl set-hostname servidor1.curso.local
